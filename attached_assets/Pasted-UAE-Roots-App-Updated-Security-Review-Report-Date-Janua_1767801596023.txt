Step 2: After confirming all data is migrated, remove CryptoJS fallback:

Replace decryptPII function (lines 102-126) with:

const decryptPII = (encrypted) => {
  if (!encrypted) return null;
  
  try {
    // Only support v2 format: v2:<iv>:<tag>:<ciphertext>
    if (!encrypted.startsWith("v2:")) {
      console.error(`[Decrypt] Rejected non-v2 format: ${encrypted.substring(0, 20)}...`);
      return "[ENCRYPTION MIGRATION REQUIRED]";
    }
    
    const parts = encrypted.split(":");
    if (parts.length !== 4) {
      console.error(`[Decrypt] Invalid v2 format structure`);
      return null;
    }
    
    const [, ivHex, tagHex, data] = parts;
    const decipher = crypto.createDecipheriv(
      "aes-256-gcm",
      DERIVED_KEY,
      Buffer.from(ivHex, "hex"),
      { authTagLength: 16 },
    );
    decipher.setAuthTag(Buffer.from(tagHex, "hex"));
    let decrypted = decipher.update(data, "hex", "utf8");
    return decrypted + decipher.final("utf8");
  } catch (error) {
    console.error(`[Decrypt] Decryption failed: ${error.message}`);
    return null; // Don't expose encrypted data
  }
};

// Cleanup orphaned photos - files in /uploads/ not referenced by any person
const cleanupOrphanedPhotos = async () => {
  try {
    const uploadedFiles = fs.readdirSync(uploadsDir);
    
    // Get all photo URLs from database
    const peopleWithPhotos = await db
      .select({ photoUrl: people.photoUrl })
      .from(people)
      .where(sql`${people.photoUrl} IS NOT NULL AND ${people.photoUrl} != ''`);
    
    const usedFilenames = peopleWithPhotos.map(p => 
      p.photoUrl.split('/').pop()
    );
    
    let deletedCount = 0;
    uploadedFiles.forEach(file => {
      // Skip hidden files and non-images
      if (file.startsWith('.') || !file.match(/\.(jpg|jpeg|png|gif|webp)$/i)) {
        return;
      }
      
      if (!usedFilenames.includes(file)) {
        const filePath = path.join(uploadsDir, file);
        fs.unlinkSync(filePath);
        deletedCount++;
        console.log(`[Photo Cleanup] Removed orphaned photo: ${file}`);
      }
    });
    
    if (deletedCount > 0) {
      console.log(`[Photo Cleanup] Removed ${deletedCount} orphaned photos`);
    }
  } catch (error) {
    console.error('[Photo Cleanup] Error:', error.message);
  }
};

// Run photo cleanup 2 hours after startup, then daily
setTimeout(() => {
  cleanupOrphanedPhotos();
  setInterval(cleanupOrphanedPhotos, 24 * 60 * 60 * 1000);
}, 2 * 60 * 60 * 1000);
