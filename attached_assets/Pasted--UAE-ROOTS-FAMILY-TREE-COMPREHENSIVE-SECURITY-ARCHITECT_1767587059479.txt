1. ENCRYPTION KEY REUSE - JWT_SECRET FALLBACK
Location: server/index.js:61

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY || JWT_SECRET;

Risk: If ENCRYPTION_KEY is not set, the application reuses JWT_SECRET for encrypting PII data (phone numbers, emails, identification numbers). This creates:

Key Compromise Cascade: If JWT secret is compromised, ALL encrypted PII is exposed
Cryptographic Key Separation Violation: Same key for authentication and data encryption
Compliance Violation: GDPR/data protection regulations require separate encryption keys
Impact: Complete exposure of all PII if JWT secret leaks (session hijacking + data breach)

Remediation:

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY;
const JWT_SECRET = process.env.JWT_SECRET;

if (!ENCRYPTION_KEY || !JWT_SECRET) {

  process.exit(1);
}

Location: server/index.js:1820-1862

Current Issues:

TOCTOU Vulnerability: File is written to disk BEFORE magic byte verification
Race Condition: Window between upload and validation where malicious file exists
Insufficient MIME Type Validation: Only checks claimed MIME, not actual content
Current Code Flow:

1. Multer accepts file based on claimed mimetype (line 316)
2. File written to disk (line 301-309)
3. Magic bytes verified AFTER file exists (line 1835)
4. If invalid, file deleted (line 1837)

Exploit Scenario:

Attacker uploads PHP shell with "image/png" MIME type
File briefly exists on disk as UUID.png
If attacker can trigger execution during validation window → RCE
Even after deletion, file fragments may persist
Remediation:

// server/index.js - Replace upload.single() with in-memory storage
const upload = multer({

    const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
    if (allowedTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('نوع الملف غير مدعوم'));

Attacker can generate unlimited JWT tokens
Even if user logs out, attacker can re-authenticate with stolen Firebase token
Remediation:

// Track used Firebase tokens to prevent reuse
const usedTokens = new Set();

app.post("/api/auth/token", async (req, res) => {
  const { userId, firebaseIdToken } = req.body;
  
  // Verify token hasn't been used before
  const tokenHash = crypto.createHash('sha256').update(firebaseIdToken).digest('hex');
  if (usedTokens.has(tokenHash)) {
    await logAudit(userId, 'login_failed', 'auth', null, 
      { reason: 'token_reuse_detected' }, req);
    return res.status(401).json({ error: 'Token already used' });
  }
  
  const decodedToken = await admin.auth().verifyIdToken(firebaseIdToken, true); // checkRevoked = true
  
  if (decodedToken.uid !== userId) {
    return res.status(401).json({ error: "Token mismatch" });
  }
  
  // Mark token as used
  usedTokens.add(tokenHash);
  
  // Clean up expired tokens every hour
  setTimeout(() => usedTokens.delete(tokenHash), 3600000);
  
  const token = jwt.sign({ userId }, JWT_SECRET, { expiresIn: "24h" });
  res.cookie("auth_token", token, COOKIE_OPTIONS);
  res.json({ token, userId });
});

// server/index.js - Add data retention cleanup

const DATA_RETENTION_YEARS = 10; // Configure based on legal requirements

const cleanupOldData = async () => {
  try {
    const cutoffDate = new Date();
    cutoffDate.setFullYear(cutoffDate.getFullYear() - DATA_RETENTION_YEARS);
    
    // Find trees not accessed in retention period
    const inactiveTrees = await db
      .select()
      .from(trees)
      .where(lt(trees.createdAt, cutoffDate));
    
    // Notify tree owners before deletion (30-day notice)
    for (const tree of inactiveTrees) {
      const notificationSent = await db
        .select()
        .from(auditLogs)
        .where(and(
          eq(auditLogs.resourceId, tree.id.toString()),
          eq(auditLogs.action, 'retention_warning')
        ));
      
      if (notificationSent.length === 0) {
        // Send warning email (implement email service)
        await logAudit(tree.createdBy, 'retention_warning', 'tree', tree.id, {
          message: 'Tree will be deleted in 30 days due to inactivity'
        }, { ip: 'system', headers: {} });
      } else {
        // Check if 30 days passed since warning
        const warningDate = new Date(notificationSent[0].createdAt);
        const thirtyDaysAgo = new Date();
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
        
        if (warningDate < thirtyDaysAgo) {
          // Delete tree and all data
          await db.delete(relationships).where(eq(relationships.treeId, tree.id));
          await db.delete(people).where(eq(people.treeId, tree.id));
          await db.delete(editHistory).where(eq(editHistory.treeId, tree.id));
          await db.delete(trees).where(eq(trees.id, tree.id));
          
          await logAudit(tree.createdBy, 'retention_deletion', 'tree', tree.id, {
            reason: 'Data retention policy'
          }, { ip: 'system', headers: {} });
        }
      }
    }
  } catch (error) {
    console.error('[Retention Cleanup] Error:', error.message);
  }
};

class ErrorBoundary extends React.Component {
  state = { hasError: false, error: null, errorInfo: null };
  
  static getDerivedStateFromError(error) {
    return { hasError: true };
  }
  
  componentDidCatch(error, errorInfo) {
    this.setState({ error, errorInfo });
    
    // Log to backend
    fetch('/api/client-errors', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        error: error.toString(),
        stack: error.stack,
        componentStack: errorInfo.componentStack,
        url: window.location.href,
        userAgent: navigator.userAgent
      })
    });
  }
  
  render() {
    if (this.state.hasError) {
      return (
        <div className="error-container">

          </Button>

  }
}

// server/websocket.js
import { WebSocketServer } from 'ws';

export const setupWebSocket = (server) => {
  const wss = new WebSocketServer({ server });
  
  const treeRooms = new Map(); // treeId -> Set of connections
  
  wss.on('connection', (ws, req) => {
    let currentTreeId = null;
    
    ws.on('message', (data) => {
      const message = JSON.parse(data);
      
      switch (message.type) {
        case 'join_tree':
          currentTreeId = message.treeId;
          if (!treeRooms.has(currentTreeId)) {
            treeRooms.set(currentTreeId, new Set());
          }
          treeRooms.get(currentTreeId).add(ws);
          break;
        
        case 'tree_update':
          // Broadcast to all clients viewing this tree
          const room = treeRooms.get(currentTreeId);
          if (room) {
            room.forEach(client => {
              if (client !== ws && client.readyState === WebSocket.OPEN) {
                client.send(JSON.stringify({
                  type: 'tree_updated',
                  data: message.data
                }));
              }
            });
          }
          break;
      }
    });
    
    ws.on('close', () => {
      if (currentTreeId && treeRooms.has(currentTreeId)) {
        treeRooms.get(currentTreeId).delete(ws);
      }
    });
  });
};
